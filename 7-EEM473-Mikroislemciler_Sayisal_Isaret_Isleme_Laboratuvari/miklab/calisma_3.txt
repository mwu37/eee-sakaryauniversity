#include 	<aduc841.h>
VLOW 	EQU 	40H
VHIGH 	EQU 	41H
	ORG 	0000H
	SJMP	AYARLAR
	ORG 	002BH
	SJMP 	T2KESME
AYARLAR:
	MOV 	DACCON,#00101101B
	MOV 	T2CON,#00000000B
	MOV 	DPTR,#54473D 	;1 MS
	MOV 	TL2,#DPL		
	MOV 	TH2,#DPH
	MOV 	RCAP2L,#DPL
	MOV 	RCAP2H,#DPH
	
	MOV 	44H,#00H 		;200D HIGH
	MOV 	43H,#0C8H   	;200D LOW
	MOV 	VHIGH,#00H 	
	MOV 	VLOW,#00H 	
	SETB 	EADC
	SETB 	EA
	SETB 	ET2
	SETB 	TR2
SONSUZ:
	SJMP 	SONSUZ
T2KESME:
	CLR 	TF2
	CLR 	C
	MOV 	A,43H
	ADD 	A,VLOW
	MOV 	VLOW,A
	MOV 	A,44H
	ADDC 	A,VHIGH
	MOV 	VHIGH,A
	MOV 	DAC0H,VHIGH
	MOV 	DAC0L,VLOW
	RETI
	END
