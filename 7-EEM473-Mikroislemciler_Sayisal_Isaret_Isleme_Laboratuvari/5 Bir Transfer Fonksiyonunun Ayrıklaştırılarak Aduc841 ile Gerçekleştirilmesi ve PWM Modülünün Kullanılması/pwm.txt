#include<Aduc841.h>
;------------------

KATSAYI_0L EQU 20H
KATSAYI_0H EQU 21H

KATSAYI_1L EQU 22H
KATSAYI_1H EQU 23H

CSEG

ORG 00H
	
SJMP MAIN

MAIN:

MOV SP, #070H
MOV PWMCON, #00010011B
MOV KATSAYI_0L, #00H
MOV KATSAYI_0H, #36H

MOV KATSAYI_1L, #00H
MOV KATSAYI_1H, #0D8H

ACALL PWM_CIKIS_VER

ASSDF:JNB P3.2,ASSDF
SERS:JB P3.2,SERS


MOV PWM0L,#00H
MOV PWM1L, KATSAYI_1L

MOV PWM0H,#06CH
MOV PWM1H, KATSAYI_1H



ASSDF1:JNB P3.2,ASSDF1
SERS2:JB P3.2,SERS2


MOV PWM0L,#00H
MOV PWM1L, KATSAYI_1L
MOV PWM0H,#0A2H
MOV PWM1H, KATSAYI_1H






bekle:
SJMP bekle
;-----------------------
PWM_CIKIS_VER:
MOV CFG841, #00H

MOV PWM0L, KATSAYI_0L
MOV PWM1L, KATSAYI_1L
MOV PWM0H, KATSAYI_0H
MOV PWM1H, KATSAYI_1H
RET

END
