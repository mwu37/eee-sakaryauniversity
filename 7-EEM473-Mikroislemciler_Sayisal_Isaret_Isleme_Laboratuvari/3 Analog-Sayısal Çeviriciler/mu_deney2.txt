#include 	<aduc841.h>
	ORG 	0000H
	SJMP 	BASLA
	ORG 	0033H
	SJMP 	OKU_YAZ
	
BASLA:
	MOV 	ADCCON1,#11111110B 	;ADC POWER ON , HARICI REF , MCLK = 2 , TF2
	MOV 	ADCCON2,#01H		;ADC1 KANAL SECILDI
	MOV 	DACCON,#01101101B	;DAC0 12-BIT HARICI REF 5V
	MOV 	T2CON,#00H			;16-BIT YENIDEN YUKLEMELI
	MOV 	RCAP2L,#0FDH		;T=5MS
	MOV 	RCAP2H,#027H		;SFR LERE YUKLANIYOR
	MOV 	TL2,#0FDH
	MOV 	TH2,#027H
	SETB 	EADC
	SETB 	EA
	SETB 	TR2
	
DUR:SJMP 	DUR

OKU_YAZ:
	CLR 	TF2
	MOV 	DAC0H,ADCDATAH
	MOV 	DAC0L,ADCDATAL
	RETI
	
	END
